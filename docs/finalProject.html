1. Getting Started

1.1 Get the Data

We will get our data from Kaggle. You might need a Kaggle account, but it is worth getting due to all of the datasets available. All we need to do is download the .csv file and read it. This data gives us data collected about the pollutants, NO2, O3, SO2, and CO. All of these are within the U.S between 2000 - 2016. However, data is not for every state, and is not consistent. So sometimes we will have missing data for one year in a certain state. It is worth noting that this data set has nearly about 1.7 million entries, so some operation will be slow.

# Data from: https://www.kaggle.com/sogun3/uspollution/downloads/uspollution.zip/1
csv_file <- "pollution_us_2000_2016.csv"
tidy_df <- read_csv(csv_file)

1.2 Tidy and Clean the Data

Now we can start to tidy and clean our data. First we will remove all of the rows that are unnecessary. We will also create create a year column. This is to reduce the processing time for the future, due to the large data set.

# Remove the columns: row number ("X1"), State Code, Country Code, Site Num, Address, and County
tidy_df <- tidy_df[-c(1:5,7)]

# Now delete all of the Units columns
# NO2 = parts per billion
# O3  = parts per million
# SO2 = parts per billion
# CO  = parts per million
tidy_df <- tidy_df[-c(5,10,15,20)]

# Remove the hour of the max value columns
tidy_df <- tidy_df[-c(4,8,12,16)]

# rename date column a DATE variable
colnames(tidy_df)[3] <- "Date"
tidy_df <- type_convert(tidy_df, col_types = cols("Date" = col_datetime("%Y-%m-%d")))

# Add a year column to speed up data visualization
tidy_df$year <- substring(tidy_df$Date,1,4)

# Add a month column
tidy_df$month <- substring(tidy_df$Date,6,7)

server <- function(input, output, session){
  sliderValues <- reactive({
    data.frame(
      Name = c("Year"),
      Value = as.character(c(input$Year), stringsAsFactors = FALSE))
  })
  
  output$map <- renderLeaflet({
    map <- leaflet(states) %>%
      addProviderTiles("MapBox", options = providerTileOptions(
        id = "mapbox.light",
        accessToken = Sys.getenv('MAPBOX_ACCESS_TOKEN'))) %>%
      addPolygons(
        fillColor = ~pal(states$density),
        weight = 2,
        opacity = 1,
        color = "white",
        dashArray = "3",
        fillOpacity = 0.7,
        highlight = highlightOptions(
          weight = 5,
          color = "#666",
          dashArray = "",
          fillOpacity = 0.7,
          bringToFront = TRUE),
        label = labels,
        labelOptions = labelOptions(
          style = list("font-weight" = "normal", padding = "3px 8px"),
          textsize = "15px",
          direction = "auto")) %>%
      addLegend(pal = pal, values = ~density, opacity = 0.7, title = NULL,
                position = "bottomright") %>%
      addMarkers(markers_df$lng, markers_df$lat, popup = markers_df$City, icon = myIcon) %>%
      setView(lat=39.508577, lng=-98.322468, zoom=4)
    
  })
  
  observe({ 
    # Set zoom and center map
    zoom <- reactive({
      ifelse(is.null(input$map01_zoom),4,input$map01_zoom)
    })
    center <- reactive({
        if(is.null(input$map01_center)){
          return(c(39.508577, -98.322468))
          }else{
            return(input$map01_center)
          }
    })
    
    # Get new data based off of year value
    temp_df <- tidy_df %>%
      filter(year == input$Year) %>%
      select(State, `NO2 AQI`) %>%
      group_by(State) %>%
      summarise(x = mean(`NO2 AQI`,na.rm = TRUE)) %>%
      right_join(states_name, by = "State")
    colnames(temp_df)[2] <- "density"
    # Set new density values
    states$density <- temp_df$density
    
    # Update the map
    output$map <- renderLeaflet({
      map <- leaflet(states) %>%
        addProviderTiles("MapBox", options = providerTileOptions(
          id = "mapbox.light",
          accessToken = Sys.getenv('MAPBOX_ACCESS_TOKEN'))) %>%
        addPolygons(
          fillColor = ~pal(states$density),
          weight = 2,
          opacity = 1,
          color = "white",
          dashArray = "3",
          fillOpacity = 0.7,
          highlight = highlightOptions(
            weight = 5,
            color = "#666",
            dashArray = "",
            fillOpacity = 0.7,
            bringToFront = TRUE),
          label = labels,
          labelOptions = labelOptions(
            style = list("font-weight" = "normal", padding = "3px 8px"),
            textsize = "15px",
            direction = "auto")) %>%
        addLegend(pal = pal, values = ~density, opacity = 0.7, title = NULL,
                  position = "bottomright") %>%
        addMarkers(markers_df$lng, markers_df$lat, popup = markers_df$City, icon = myIcon) %>%
        setView(lat=center()[1], lng=center()[2], zoom=zoom())
      
    })
    
  })
  
}

# 14 States with complete data between 2000 - 2015
# tidy_df %>% filter(State %in% train_states) %>% count <= size of data set = 1306313
train_states <- c("Arizona", "California", "Colorado", "District Of Columbia", "Florida", "Illinois", "Indiana", "Kansas", "Louisiana", "New York", "North Carolina", "Oklahoma", "Pennsylvania", "Texas")

temp_df1 <- tidy_df %>%
  filter((State %in% train_states) & year == 2015) %>%
  select(State, `NO2 AQI`) %>%
  group_by(State) %>%
  summarise(x = mean(`NO2 AQI`, na.rm = TRUE))

temp_df2 <- tidy_df %>%
  filter((State %in% train_states) & year == 2014) %>%
  select(State, `NO2 AQI`) %>%
  group_by(State) %>%
  summarise(x = mean(`NO2 AQI`, na.rm = TRUE))

outcome_df <- inner_join(temp_df1, temp_df2, by = "State")
colnames(outcome_df)[2] <- "year2015"
colnames(outcome_df)[3] <- "year2014"
outcome_df <- outcome_df %>%
  mutate(diff = year2015 - year2014) %>%
  mutate(direction = ifelse(diff > 0, "up", "down")) %>%
  select(State, direction)
# Create predictor_df X_i to build a classifier for the outcome
predictor_df <- tidy_df %>%
  filter((State %in% train_states) & year <= 2015) %>%
  select(State, year, `NO2 AQI`) %>%
  group_by(State, year) %>%
  summarise(x = mean(`NO2 AQI`, na.rm = TRUE))
colnames(predictor_df)[3] <- "NO2AQI"

wide_df <- predictor_df %>%
  ungroup() %>%
  select(State, year, NO2AQI) %>%
  tidyr::spread(year, NO2AQI)

final_df <- wide_df %>%
  inner_join(select(outcome_df, State, direction), by="State") %>%
  mutate(direction=factor(direction, levels=c("down", "up")))
set.seed(123456789)
test_rf_df <- final_df %>%
  group_by(direction) %>%
  sample_frac(.5) %>%
  ungroup()

train_rf_df <- final_df %>%
  anti_join(test_rf_df, by="State")

rf_fit <- train(direction~.,
                data = train_rf_df %>% select(-State),
                method="rf",
